# create pip for each vm
# Instance #1
# Instance Name: aks-system-34603732-vmss_0
# Computer Name: aks-system-34603732-vmss000000
# Private IPv4: 172.16.1.5
# Public IPv6: fdf0:1e2d:3c4b::4

VM="aks-system-34603732-vmss000000" #VM Name
MC_RG="MC_island-rg_island-aks-byo_swedencentral" #MC RG
INT_PIP6="fdf0:1e2d:3c4b::4" #private ipv6 assigned to node

# general naming scheme
FE_NAME="${VM}-pip6"
BE_NAME="${VM}-BE-ipv6"
INBOUND_RULE_NAME="${VM}-in-rule"
OUT_RULE_NAME="${VM}-out-rule"

# creating a backend pool with vm ip
az network lb address-pool create \
  --resource-group "$RG" \
  --lb-name "$LB_NAME" \
  --name "$BE_NAME" \
  --vnet "$VNET" \
  --backend-addresses "[{name:addr1,ip-address:'$INT_PIP6'}]"

# creating a pip.
az network public-ip create \
  -g "$RG" \
  -n "$FE_NAME" \
  -l "$LOCATION" \
  --sku Standard \
  --allocation-method Static \
  --version IPv6

# assigning the pip to lb frontend
az network lb frontend-ip create \
 -g "$RG" \
 -n "$FE_NAME" \
 --lb-name "$LB_NAME" \
 --public-ip-address "$FE_NAME"

# this will enable outbound on ipv6
az network lb outbound-rule create \
  -g "$RG" \
  --lb-name "$LB_NAME" \
  --name "$OUT_RULE_NAME" \
  --frontend-ip-configs "$FE_NAME" \
  --backend-address-pool "$BE_NAME" \
  --protocol All

# we cannot create a rule for all ports and protocols on a public lb
# for example we create tcp for all incoming port 8080.
az network lb inbound-nat-rule create \
 -g $RG \
 --lb-name $LB_NAME \
 --name  "$INBOUND_RULE_NAME" \
 --protocol TCP \
 --frontend-ip-name "$FE_NAME" \
 --frontend-port-range-start 8080 \
 --frontend-port-range-end 8080 \
 --floating-ip false \
 --backend-port 8080 \
 --backend-address-pool "$BE_NAME"
